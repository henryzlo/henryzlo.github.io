CS310 Example of a private inner class providing an iterator

 

// Weiss example of pg. 578-579 (522-523) converted to use generics

import java.util.Iterator;

 

public class MyContainer<T> {

       // Note: can't use "T[] items = new T[5]" here

       // see "Notes on Generics" linked from the class web page

       // but could put:

       // @SuppressWarnings("unchecked")

       // private T[] items = (T[])(new Object[5]);

       private Object[] items = new Object[5];                        

       private int size = 0;

 

       @SuppressWarnings("unchecked")  // for cast at return

       public T get(int idx) {

              if (idx < 0 || idx >= size)

                     throw new ArrayIndexOutOfBoundsException("Index " + idx + "; size "

                                  + size);

              return (T)items[idx];

       }

 

       public boolean add(T x) {

              if (items.length == size) {

                     Object[] old = items;

                     items = new Object[items.length * 2 + 1];

                     for (int i = 0; i < size; i++)

                           items[i] = old[i];

              }

              items[size++] = x;

              return true;

       }

 

       public Iterator<T> iterator() {

              return new LocalIterator();

       }

 

       // Note: the parametrization by T of LocalIterator is there

// because this is an inner class (no “static” keyword).

       // In fact, if you put LocalIterator<T> it will cause unwanted

       // additional parametrization and not work as desired

       // and causes warning: "type T is hiding the type T"

       private class LocalIterator implements Iterator<T> {

            private int current = 0;

 

            @Override

      public boolean hasNext() {

                     return current < size;

            }

 

            @SuppressWarnings("unchecked")  // for cast at return

            @Override

            public T next() {

                     return (T)items[current++];

            }

 

            @Override

            public void remove() {

                     throw new UnsupportedOperationException();

            }

       }

 

       public static void main(String[] args) {

              MyContainer<String> strs = new MyContainer<String>();

              strs.add("Foo");

              strs.add("bar");

              Iterator<String> itr1 = strs.iterator();

              String s1 = itr1.next();

              System.out.println(s1);

       }

}
