CS310 MakeChange Example: Recursive Search with Dynamic Programming

// CS310 pa3 example, based on Weiss, pg. 331 (289).

// MakeChange: a recursive search that remembers what it finds out

// along the way (a technique known as dynamic programming.)

// Note that makeChange1 is a recursion helper.  makeChange is

// the public function, and does some setup, then calls on makeChange1

// to do the main recursion.  You need such a trick in pa3.

// Also, note how we can output reports with recursion level indicated

// by indentation, by tracking recursion level in the recursion helper.

 

import ...

public class MakeChange {

 

       // the top-level makeChange function: same functionality

       // as Weiss, pg. 331 (289)

       public static int makeChange(int[] coins, int change) {

              // for saving facts: how many coins are needed for a value

              Map<Integer, Integer> coinCount = new TreeMap<Integer, Integer>();

              // For each coin, one of it is best for its own value

              for (int i = 0; i < coins.length; i++)

                     coinCount.put(coins[i], 1);

              int minCoins = makeChange1(coins, change, coinCount, 0);

              System.out.println("countCount map = " +coinCount);

              return minCoins;

       }

 

       // helper to makeChange--do the recursion, once coinCount is set up

       // level: recursion level, for tracing execution.

       private static int makeChange1(int[] coins, int change,

                     Map<Integer, Integer> coinCount, int level) {

              int minCoins = change;

              Integer oldCoins;

 

              level++; // for trace output, track recursion level

              indentPrint(level, "makeChange1(" + change + ")"

+ (coinCount.get(change)== null? " (needs calc)":""));

   // try to look up coin count in map

              if ((oldCoins = (Integer) coinCount.get(change)) != null) {

                     return oldCoins;

              }

 

              // lookup failed, need to compute coin count

              for (int j = 1; j <= change / 2; j++) {

                     int coins1 = makeChange1(coins, j, coinCount, level);

                     int coins2 = makeChange1(coins, change - j, coinCount, level);

                     int thisCoins = coins1 + coins2;

                     if (thisCoins < minCoins)

                           minCoins = thisCoins;

              }

 

              // save newly computed coin count

              indentPrint(level, "saving final change for " + change + ": #coins = "

                           + minCoins);

              coinCount.put(change, minCoins);

              return minCoins;

       }

 

       // Print s, indented by n spaces

       private static void indentPrint(int n, String s) {

              for (int i = 0; i < n; i++)

                     System.out.print(" ");

              System.out.println(s);

       }

 

       public static void main(String[] args) {

              int coins[] = { 1, 5, 10, 25 }; // US currency

              int ncoins = makeChange(coins, 7);

              ncoins = makeChange(coins, 13);

              System.out.println("#coins for 13 = " + ncoins);

              ncoins = makeChange(coins, 63);

              System.out.println("#coins for 63 = " + ncoins);

              int coins1[] = { 1, 5, 10, 21, 25 }; // US, plus 21-cent piece

              ncoins = makeChange(coins1, 63);

              System.out.println("With 21-cent pieces, #coins for 63 = " + ncoins);

       }

}
